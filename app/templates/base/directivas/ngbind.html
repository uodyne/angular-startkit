ng-bind

<div ng-controller="MyController" >
  <span ng-bind="myData.textf()"></span>
</div>

  <script>
    angular.module("myapp", [])
    .controller("MyController", function($scope) {
      $scope.myData = {};
      $scope.myData.textf = function() { return "A text from a function"; };
    });
  </script>
